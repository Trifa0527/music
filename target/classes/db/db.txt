CREATE DATABASE MUSIC;

CREATE TABLE MUSICINFO(
	MUSICID INT(10),
	ALBUMID INT(10),
	MUSICTITLE TEXT,
	MUSICINTRO TEXT,
	MUSICARTIST TEXT,
	MUSICDATE DATE,
	PRIMARY KEY(MUSICID, ALBUMID)
);

CREATE TABLE ALBUMINFO(
	ALBUMID INT(10),
	ALBUMTITLE TEXT,
	ALBUMINTRO TEXT,
	ALBUMDATE DATE,
	PRIMARY KEY(ALBUMID)
);
