CREATE DATABASE MUSIC;

CREATE TABLE MUSICLIST(
	MUSICID INT(10),
	PRIMARY KEY(MUSICID)
);

CREATE TABLE MUSICINFO(
	MUSICID INT(10),
	ALBUMID INT(10),
	MUSICTITLE TEXT,
	MUSICINTRO TEXT,
	MUSICARTIST TEXT,
	MUSICDATE DATE,
	PRIMARY KEY(MUSICID, ALBUMID)
);

CREATE TABLE ALBUMLIST(
	ALBUMID INT(10),
	PRIMARY KEY(ALBUMID)
);

CREATE TABLE ALBUMINFO(
	ALBUMID INT(10),
	ALBUMTITLE TEXT,
	ALBUMINTRO TEXT,
	ALBUMDATE DATE,
	PRIMARY KEY(ALBUMID)
);

SELECT * FROM MUSICLIST LEFT OUTER JOIN MUSICINFO ON MUSICINFO.MUSICID = MUSICLIST.MUSICID WHERE MUSICLIST.MUSICID=?;
